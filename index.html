// Class abstrak Pembayaran
abstract class Pembayaran {
  void prosesPembayaran();
}

// Class TransferBank (Inheritance)
class TransferBank extends Pembayaran {
  String namaBank;

  TransferBank(this.namaBank);

  @override
  void prosesPembayaran() {
    print("\n=== PEMBAYARAN ===");
    print("Pembayaran melalui $namaBank berhasil.");
  }
}

// Class Penumpang
class Penumpang {
  int id;
  String nama;
  String nik;

  Penumpang(this.id, this.nama, this.nik);

  void tampilInfo() {
    print("ID        : $id");
    print("Nama      : $nama");
    print("NIK       : $nik");
  }
}

// Class Kereta
class Kereta {
  String kodeKereta;
  String namaKereta;
  int kapasitas;

  Kereta(this.kodeKereta, this.namaKereta, this.kapasitas);
}

// Class Gerbong
class Gerbong {
  int nomorGerbong;
  int kapasitas;

  Gerbong(this.nomorGerbong, this.kapasitas);
}

// Class Jadwal
class Jadwal {
  String asal;
  String tujuan;
  String tanggal;
  String jam;

  Jadwal(this.asal, this.tujuan, this.tanggal, this.jam);
}

// Class Tiket
class Tiket {
  String nomorTiket;
  double harga;
  Kereta kereta;
  Gerbong gerbong;
  Jadwal jadwal;

  Tiket(
    this.nomorTiket,
    this.harga,
    this.kereta,
    this.gerbong,
    this.jadwal,
  );

  void tampilTiket() {
    print("Nomor Tiket : $nomorTiket");
    print("Kereta      : ${kereta.namaKereta}");
    print("Gerbong     : ${gerbong.nomorGerbong}");
    print("Rute        : ${jadwal.asal} - ${jadwal.tujuan}");
    print("Harga       : Rp$harga");
  }
}

// Class Pemesanan
class Pemesanan {
  String kodePemesanan;
  Penumpang penumpang;
  Tiket tiket;

  Pemesanan(this.kodePemesanan, this.penumpang, this.tiket);

  void tampilPemesanan() {
    print("\n=== DETAIL PEMESANAN ===");
    print("Kode Pemesanan : $kodePemesanan");
    print("Nama Penumpang : ${penumpang.nama}");
    print("Nomor Tiket    : ${tiket.nomorTiket}");
    print("Kereta         : ${tiket.kereta.namaKereta}");
    print("Gerbong        : ${tiket.gerbong.nomorGerbong}");
    print("Rute           : ${tiket.jadwal.asal} - ${tiket.jadwal.tujuan}");
    print("Tanggal        : ${tiket.jadwal.tanggal}");
    print("Jam            : ${tiket.jadwal.jam}");
    print("Total Harga    : Rp${tiket.harga}");
  }
}

void main() {
  // Data kereta
  Kereta kereta = Kereta("KA001", "Sriwijaya Express", 500);

  // Data gerbong
  Gerbong gerbong1 = Gerbong(1, 10);
  Gerbong gerbong2 = Gerbong(2, 5);

  // Data jadwal
  Jadwal jadwal =
      Jadwal("Palembang", "Lampung", "20-06-2026", "08:00");

  // List pemesanan
  List daftarPemesanan = [];

  // 10 penumpang di gerbong 1
  for (int i = 1; i <= 10; i++) {
    Penumpang penumpang =
        Penumpang(i, "Penumpang $i", "167101010101000$i");

    Tiket tiket =
        Tiket("TKT00$i", 150000, kereta, gerbong1, jadwal);

    daftarPemesanan.add(
        Pemesanan("ORD00$i", penumpang, tiket));
  }

  // 5 penumpang di gerbong 2
  for (int i = 11; i <= 15; i++) {
    Penumpang penumpang =
        Penumpang(i, "Penumpang $i", "167101010101000$i");

    Tiket tiket =
        Tiket("TKT00$i", 150000, kereta, gerbong2, jadwal);

    daftarPemesanan.add(
        Pemesanan("ORD00$i", penumpang, tiket));
  }

  // Tampilkan pemesanan + pembayaran per penumpang
  for (var pemesanan in daftarPemesanan) {
    pemesanan.tampilPemesanan();

    TransferBank pembayaran = TransferBank("BRI");
    pembayaran.prosesPembayaran();
  }
}
